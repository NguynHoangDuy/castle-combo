Hire an artist for custom illustrations.

## Placeholder Generator Script

Create simple placeholder images for development:

```javascript
// placeholders.js - Run with Node.js + Canvas
const { createCanvas } = require('canvas');
const fs = require('fs');

const characters = [
  'king', 'queen', 'prince', // ... etc
];

characters.forEach(name => {
  const canvas = createCanvas(250, 300);
  const ctx = canvas.getContext('2d');

  // Background
  ctx.fillStyle = '#4a5568';
  ctx.fillRect(0, 0, 250, 300);

  // Text
  ctx.fillStyle = '#ffffff';
  ctx.font = '24px Arial';
  ctx.textAlign = 'center';
  ctx.fillText(name.toUpperCase(), 125, 150);

  // Save
  const buffer = canvas.toBuffer('image/png');
  fs.writeFileSync(`./assets/cards/characters/${name}.png`, buffer);
});
```
